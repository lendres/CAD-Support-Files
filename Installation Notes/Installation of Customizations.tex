\section{Paths}
\label{sec:paths}
\begin{numberedlist}
    \item Open the options (command \textcode{options}).
    \item Add the support file path.
    \begin{numberedlist}
    	\item Expand \textcode{File$\rightarrow$Support File Search Path}.
    	\item Add the \supportdir{} directory.
    	\item Move the entry to the top so that anything in that directory is found first.  See \figurename~\ref{fig:addsupportdirectory}.
    	\item Add an entry for each subdirectory in \blocksdir{}.
    \end{numberedlist}
    \item Add the plot style search path.
    \begin{numberedlist}
    	\item Expand \textcode{Printer Support File Path}$\rightarrow$\textcode{Printer Style Table Search Path}
    	\item Add the \plotstyledir{} directory.
    \end{numberedlist}
    \item Add the template files location.
    \begin{numberedlist}
    	% C:\Users\Lance\AppData\Local\Gstarsoft\GstarCAD\R23\en-US\template\
        % C:\Users\Lance\AppData\Local\ZWSOFT\ZWCAD\2023\en-US\Template
    	\item Expand \textcode{Template Settings}$\rightarrow$\textcode{Drawing Template File Location}
    	\item Add the \templatedir{} directory.
    \end{numberedlist}
\end{numberedlist}

\section{Command Aliases (PGP)}
\subsection{New Software Installation of Previously Install Version}
If the installation steps in \sectionname{\ref{sec:pgpinstallation}} have already been done for the version of the software that was installed (e.g. ZWCAD2023 OR ZWCAD2024) then no further installation is required.  The new \textcode{pgp} file is already installed in the \supportdir{} directory and will be found once the paths are set according to \sectionname{\ref{sec:paths}}.

\subsection{New Software Version Installation}
\label{sec:pgpinstallation}
For new software versions, the \textcode{pgp} file must be created by adding custom code to the software's \textcode{pgp} file.
\begin{numberedlist}
    \item Find the software's \textcode{pgp} file.  Example file location:
    \begin{plainlist}
    	\item \textcode{C:\tbs{}Users\tbs{}$<$user$>$\tbs{}AppData\tbs{}Roaming\tbs{}ZWSOFT\tbs{}ZWCAD\tbs{}2024\tbs{}en-US\tbs{}Support\tbs{}}
    \end{plainlist}
    \item Copy it to the \supportdir{} directory.
    \item Copy of contents of the \userdir\tbs\textcode{user.pgp} to the bottom of the software's \textcode{pgp} file.  The aliases at the bottom will override those at the top.
    \item Save and close the file.
    \item If the software is open, close it and restart it.
\end{numberedlist}

\section{Lisp}
\subsection{New Software Installation of Previously Install Version}
If the installation steps in \sectionname{\ref{sec:lispinstallation}} have already been done for the version of the software that was installed (e.g. ZWCAD2023 OR ZWCAD2024) then no further installation is required.  The new \textcode{lsp} file is already installed in the \supportdir{} directory and will be found once the paths are set according to \sectionname{\ref{sec:paths}}.

\subsection{New Software Version Installation}
For new software versions, the \textcode{lsp} file must be created by adding custom code to the software's \textcode{lsp} file.
\label{sec:lispinstallation}
\begin{numberedlist}
    \item Find the software's \textcode{\textit{document}.lsp}.  Examples include:
    \begin{bulletedlist}
    	\item \textcode{acaddoc.lsp}
    	\item \textcode{C:\tbs{}Program Files\tbs{}Gstarsoft\tbs{}GstarCAD2023\tbs{}Support\tbs{}gcad2023doc.lsp}
    	\item \textcode{C:\tbs{}Program Files\tbs{}ZWSOFT\tbs{}ZWCAD 2024\tbs{}Support\tbs{}en-US\tbs{}ZWCAD2024doc.lsp}
    \end{bulletedlist}
    \item Copy it to the \supportdir{} directory.
    \item Edit the file to add the following the end (just before the file \textcode{(princ)} function):
    \begin{plainlist}
    	\item \textcode{; Added to support autoload of custom lisp files.}
		\item \textcode{(load "userdoc.lsp")}
    \end{plainlist}
    \item Save and close the file.
    \item If the software is open, close it and restart it.
\end{numberedlist}

\section{Menus}
\subsection{Load a Partial Menu}
The \textcode{CTRL-D} ortho-toggle is located in the \textcode{lae.mnu} file.
\begin{bulletedlist}
	\item \textcode{MENULOAD}$\rightarrow$lae.mnu (or lae.cuix).
\end{bulletedlist}
The custom menu files are located in the \userdir{}.

\subsection{Set Osnap Menu}
\begin{numberedlist}
	\item Open the \textit{Customize User Interface} (use the command \textcode{CUI}).
	\item Go to \textcode{GCAD/ACAD}$\rightarrow$\textcode{Mouse Buttons}$\rightarrow$\textcode{Shift+Click}$\rightarrow$\textit{Button 2: Snap Button}.
	\item Change \textcode{Macro} from \textcode{\$P0=SNAP \$p0=*} to \textcode{\$P0=lae.SNAP \$p0=*}.
\end{numberedlist}
